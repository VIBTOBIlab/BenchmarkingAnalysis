03. Tutorial on how to study the variation of AUC-ROC score at varying sequencing depth

Edoardo Giuili

2025-09-24

Introduction

In this tutorial we want to show how to analyse the robustness of the tools when increasing/decreasing the sequencing depth of the mixtures. Particularly, we will use the AUC-ROC score and look whether this score drastically changes.

Import the useful functions and the metrics

source("R/compute_auc.R")

# RMSE
rmse <- function(actual, predicted) {
  round(sqrt(mean((actual - predicted)^2)),4)
}
# Normalized RMSE (NRMSE)
nrmse <- function(actual, predicted) {
    round(sqrt(mean((actual - predicted))^2)/mean(actual+10e-8),4)
}
# Spearman's rank correlation coefficient (SCC)
scc <- function(actual, predicted) {
  cor(actual, predicted, method = "spearman", use = "pairwise.complete.obs")
}
roc.obj <- function(true_labels, predicted_scores) {
  true_labels[true_labels>0] <- 1
  roc_obj <- roc(true_labels, predicted_scores, quiet = TRUE)
  return(roc_obj)
}

WGBS

We will start analysing the WGBS in silico mixtures, to then proceed to the RRBS datasets. First, we will subset the results to only keep the WGBS samples.

bench_wgbs <- subset(bench.res,bench.res$seq_method=="wgbs")

We can now compute the AUC-ROC score. Differently from the 01_ tutorial, here we will use the raw AUC-ROC score without min-max scaling it across deconvolution tools. This is necessary since we want to check, within each deconvolution tool, whether the performance changes varying the sequencing depth.

# Create a new dataset dataframe where to store AUC-ROC results
aucroc_data <- data.frame()
# Select the samples with tumoral fractions > 0%
unique_fractions <- unique(bench_wgbs$expected_tf)
unique_fractions <- subset(unique_fractions, unique_fractions != 0)

# Loop over the type of tumors
for (type in unique(bench_wgbs$tumor_type)) {
  
  # Loop over sequencing depths
  filt_tum <- subset(bench_wgbs, bench_wgbs$tumor_type == type)
  for (seqdepth in unique(filt_tum$seq_depth)) {
    
    # Loop over the DMR tools    
    filt_depth <- subset(filt_tum, filt_tum$seq_depth == seqdepth)
    for (dmrtool in unique(filt_depth$dmr_tool)) {
      
      # Loop over the deconvolution tools
      for (deconv in unique(filt_depth$deconv_tool)) {
  
        # Loop over the tumoral fractions specified above
        for (fraction in unique_fractions) {
          filt_df <- filt_depth %>%
            filter(expected_tf %in% c(0,fraction) & 
                   dmr_tool == dmrtool &
                   deconv_tool == deconv)

          # Compute the AUC-ROC score per combination and concatanate it to the final dataframe
          auc_single_combination <- compute_auc(filt_df, fraction, deconv)
          aucroc_data <- rbind(aucroc_data, auc_single_combination)
        }
      }
    }
  }
}


auc_wgbs <- aucroc_data %>%
    group_by(deconv_tool, dmr_tool, tumor_type, seq_depth, expected_tf) %>%
    summarize(AUC=mean(auc),.groups = "drop") %>%
    group_by(deconv_tool, dmr_tool, tumor_type, seq_depth) %>%
    summarize(AUC=mean(AUC),.groups = "drop")

We can now plot the AUC-ROC score distribution using the boxplots across the different sequencing depths, for each tool and for each different tumor type.

# Create a directory to save the plots (optional)
output_dir <- paste0(dir_path,"plots/03_seqdepth_robustness")
if (!dir.exists(output_dir)) dir.create(output_dir)

# Order the seq depths levels
auc_wgbs$seq_depth <- factor(auc_wgbs$seq_depth, levels = c("62M", "183M", "310M", "620M"))

# Define comparisons
my_comparisons <- list(
  c("62M", "183M"),
  c("62M", "310M"),
  c("62M", "620M")
)

# Loop over the tumor types
for (tum in unique(auc_wgbs$tumor_type)) {
  
  # Subset the dataframe on the tumor type
  df <- subset(auc_wgbs, auc_wgbs$tumor_type==tum)
  
  # Plot with adjusted p-values (FDR) and significance only if p.adj < 0.05
  p1 <- ggplot(df, aes(x = seq_depth, y = AUC, fill = seq_depth)) +
    geom_boxplot(outlier.shape = NA) +
    facet_wrap(~ deconv_tool, ncol = 5) +
    stat_compare_means(
      method = "wilcox.test", comparisons = my_comparisons,
      p.adjust.method = "BH", label = "adj.p", hide.ns = F,                    
      method.args = list(alternative="less"),
      label.y = c(1, 1.05, 1.10), size = 3
    ) +
    labs(
      title = paste0("WGBS (",tum,") dataset"),
      x = "Sequencing Depth",
      y = "AUC"
    ) +
    theme_classic() +
    theme(axis.text.x = element_blank(),
          axis.ticks.x = element_blank(),
          panel.grid.major.y = element_line(color = "grey90"),
          panel.grid.minor.y = element_line(color = "grey90")) +
    scale_fill_manual(values = c("62M"="#FAEBDDFF","183M"="#F69C73FF","310M"="#A11A5BFF","620M"="#03051AFF")) +
    ylim(0.5,1.2)
  
  # Save the plot as an image
  output_file <- file.path(output_dir, paste0("robustness_AUC_WGBS_",tum,".svg"))
  ggsave(output_file, plot = p1, width = 8, height = 6) 
}
